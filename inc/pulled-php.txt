
/**
 * Enqueue scripts and styles
 */
function wpblockshop_svg_hero_scripts() {

	wp_enqueue_style( '' );
	wp_enqueue_script( 'gasp' );

	wp_enqueue_style( 'wpblockshop-svg-hero-style', get_stylesheet_uri() );
	wp_enqueue_script( 'wpblockshop-svg-hero', get_stylesheet_uri() );
}
add_action( 'wp_enqueue_scripts', 'wpblockshop_svg_hero_scripts' );
